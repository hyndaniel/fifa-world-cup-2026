```python
# backend/results.py
"""从 sporttery 开奖接口抓 FT 终比分。key 天然 = zucai_num(与预测同键)。"""
from __future__ import annotations
from dataclasses import dataclass
import httpx

DEFAULT_API = "https://webapi.sporttery.cn/gateway/jc/football/getMatchResultV1.qry"  # 以 Task1 notes 为准
DEFAULT_UA = "Mozilla/5.0 (iPhone; CPU iPhone OS 16_0 like Mac OS X) Mobile/15E148"
REFERER = "https://m.sporttery.cn/"


@dataclass(frozen=True)
class MatchResult:
    zucai_num: str
    home_goals: int
    away_goals: int
    finished: bool


def _to_int(v) -> int | None:
    if v is None:
        return None
    s = str(v).strip()
    if not s:
        return None
    try:
        return int(float(s))
    except (TypeError, ValueError):
        return None


def _rows(data: dict) -> list[dict]:
    """取响应里的比赛列表;sporttery 常见 value 为 list 或 value.<key> 为 list。"""
    value = (data or {}).get("value")
    if isinstance(value, list):
        return value
    if isinstance(value, dict):
        for v in value.values():
            if isinstance(v, list):
                return v
    return []


def parse_results(data: dict) -> list[MatchResult]:
    out: list[MatchResult] = []
    for m in _rows(data):
        num = (m.get("matchNumStr") or "").strip()
        hg = _to_int(m.get("homeAllScore"))   # Task1 notes 对齐真实字段名
        ag = _to_int(m.get("awayAllScore"))
        finished = bool(num) and hg is not None and ag is not None
        if not num:
            continue
        out.append(MatchResult(num, hg or 0, ag or 0, finished))
    return out
```

```python
def fetch_results(cfg: dict | None = None, timeout: float = 30.0) -> list[MatchResult]:
    rcfg = (cfg or {}).get("results", {}) if cfg else {}
    api = rcfg.get("api") or DEFAULT_API
    ua = rcfg.get("ua") or DEFAULT_UA
    proxy = rcfg.get("proxy") or None
    headers = {"User-Agent": ua, "Referer": REFERER, "Accept": "application/json"}
    params = rcfg.get("params") or {"matchPage": "1", "pageNo": "1", "pageSize": "50"}  # Task1 notes 对齐
    client_kwargs: dict = {"timeout": timeout}
    if proxy:
        client_kwargs["proxy"] = proxy
    with httpx.Client(**client_kwargs) as client:
        resp = client.get(api, params=params, headers=headers)
        resp.raise_for_status()
        return parse_results(resp.json())
```

```python
# backend/mech_tag.py
"""机械对错标:三方 had 预测 argmax vs 实际 outcome,纯事实、无 LLM。"""
from __future__ import annotations
import sqlite3
from backend.baseline import baseline_market, HAD_CFG, get_v1, get_v2_prediction

def _argmax_had(probs: dict | None) -> str | None:
    if not probs:
        return None
    items = [(k, probs.get(k)) for k in ("h", "d", "a") if probs.get(k) is not None]
    if not items:
        return None
    return max(items, key=lambda kv: kv[1])[0]

def _actual(cache_path: str, match_key: str) -> str | None:
    conn = sqlite3.connect(cache_path)
    row = conn.execute("SELECT outcome FROM match_results WHERE match_key=?", (match_key,)).fetchone()
    conn.close()
    return row[0] if row else None

def _mark(pred: str | None, actual: str) -> str:
    if pred is None:
        return "—"
    return "✅" if pred == actual else "❌"

def mech_tags(cache_path: str, match_key: str) -> dict:
    actual = _actual(cache_path, match_key)
    if actual is None:
        return {"match_key": match_key, "actual": None, "v1": "—", "v2": "—", "market": "—"}
    # market
    mkt = baseline_market(cache_path, match_key, HAD_CFG)
    mkt_pred = _argmax_had(getattr(mkt, "baseline", None) or (mkt or {}).get("baseline") if mkt else None)
    # v2
    v2 = get_v2_prediction(cache_path, match_key)
    v2_pred = _argmax_had((v2 or {}).get("had"))
    # v1
    v1 = get_v1(cache_path, match_key)
    v1_pred = _argmax_had((v1 or {}).get("probs"))
    return {
        "match_key": match_key, "actual": actual,
        "v1": _mark(v1_pred, actual), "v2": _mark(v2_pred, actual), "market": _mark(mkt_pred, actual),
    }
```

```python
# tools/backfill_results.py
"""赛果回填闭环:抓终比分 → record_result → 机械对错标 → 重跑跑分卡。--once 供 launchd。"""
from __future__ import annotations
import sys, argparse, pathlib
from backend.results import fetch_results, MatchResult
from backend.baseline import record_result
from backend.mech_tag import mech_tags

REPO = pathlib.Path(__file__).resolve().parents[1]
DEFAULT_CACHE = str(REPO / ".cache" / "odds_cache.db")
DEFAULT_TAGS_OUT = str(REPO / "reports" / "复盘对错标.md")


def backfill(cache_path: str, results: list[MatchResult]) -> list[str]:
    done = []
    for r in results:
        if not r.finished:
            continue
        record_result(cache_path, r.zucai_num, r.home_goals, r.away_goals)
        done.append(r.zucai_num)
    return done


def write_mech_tags(cache_path: str, keys: list[str], out_path: str) -> None:
    if not keys:
        return
    p = pathlib.Path(out_path)
    new = not p.exists()
    lines = []
    if new:
        lines.append("# 复盘对错标(机械·纯事实)\n\n| ts | 场次 | 实际 | v1 | v2 | 市场 |\n|---|---|---|---|---|---|\n")
    import datetime as _dt  # 仅取本地时间戳,允许(非测试路径)
    ts = _dt.datetime.now().strftime("%Y-%m-%d %H:%M")
    for k in keys:
        t = mech_tags(cache_path, k)
        lines.append(f"| {ts} | {k} | {t['actual']} | {t['v1']} | {t['v2']} | {t['market']} |\n")
    with p.open("a", encoding="utf-8") as f:
        f.write("".join(lines))


def main(argv=None) -> int:
    ap = argparse.ArgumentParser()
    ap.add_argument("--once", action="store_true", help="单次执行(launchd)")
    ap.add_argument("--cache", default=DEFAULT_CACHE)
    ap.add_argument("--tags-out", default=DEFAULT_TAGS_OUT)
    args = ap.parse_args(argv)
    try:
        results = fetch_results()
    except Exception as e:
        print(f"[backfill] 抓终比分失败: {e}", file=sys.stderr)
        return 1
    keys = backfill(args.cache, results)
    if not keys:
        print("[backfill] 无新赛果可录")
        return 0
    write_mech_tags(args.cache, keys, args.tags_out)
    print(f"[backfill] 已录 {len(keys)} 场: {keys}")
    # 重跑跑分卡
    from tools import v2_report
    v2_report.main([])
    return 0


if __name__ == "__main__":
    raise SystemExit(main())
```
