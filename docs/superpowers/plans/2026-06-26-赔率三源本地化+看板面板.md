# 赔率三源本地化 + 看板赔率面板 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 把竞彩/欧盘/Poly 三源刷新收敛成本地一个 `refresh_all.py`(Poly 经 7897 代理本地直抓),并在 HK 看板每张决策卡里显示三源现价+变化+分歧。

**Architecture:** approach A——本地 `refresh_all` 抓三源→写本地缓存→按 label 对齐到 `zucai_num`、算变化与分歧→POST 给 HK 两端点(竞彩 raw 复用 `/api/ingest/zucai`,赔率面板用新 `/api/ingest/odds`)。HK 只按 `match_key` 存最新一份、join 进决策卡,前端渲染可折叠面板。

**Tech Stack:** Python 3 stdlib(urllib/sqlite3)、FastAPI(backend/web.py)、原生 JS 前端。无新第三方依赖。

## Global Constraints

- 纯标准库,不引第三方(采集脚本沿用 urllib/sqlite3)。
- 竞彩源保持 sporttery(官方价,had/hhad/ttg);**不**用 500 替代。
- 三源对齐键 = label `"home_cn vs away_cn"`(三源 fetch item 的 label 同格式);看板 `match_key` = 竞彩 `zucai_num`。
- best-effort:某源失败 → 标 `stale`/省略,其余照常;缺源明说不假算。
- 代理默认 `http://127.0.0.1:7897`,可由 `WC_HTTPS_PROXY` 覆盖。
- HK 推送地址/密码读环境变量 `WC_INGEST_URL`/`WC_INGEST_PW`/`WC_INGEST_USER`(默认 admin),镜像 `collect_zucai.py`。
- 分歧徽标阈值:任一选项 `|竞彩去水 − 欧盘去水| ≥ 3pp`。
- 提交粒度:每个 Task 末尾 commit。

```python
    def save_odds(self, items):
        """逐条按 match_key upsert 赔率面板 payload(替换语义); 缺 match_key 跳过; 返回写入数。"""
        ts = _now_bj()
        n = 0
        with self._conn() as conn:
            for it in items or []:
                if not isinstance(it, dict):
                    continue
                mk = it.get("match_key")
                if not mk:
                    continue
                conn.execute(
                    """INSERT INTO odds (match_key, ts, payload_json) VALUES (?,?,?)
                       ON CONFLICT(match_key) DO UPDATE SET
                         ts=excluded.ts, payload_json=excluded.payload_json""",
                    (mk, ts, json.dumps(it, ensure_ascii=False)),
                )
                n += 1
        return n

    def get_odds(self):
        """返回 {match_key: payload_dict}; 损坏行跳过。"""
        with self._conn() as conn:
            rows = conn.execute("SELECT match_key, payload_json FROM odds").fetchall()
        out = {}
        for r in rows:
            try:
                out[r["match_key"]] = json.loads(r["payload_json"])
            except (TypeError, ValueError):
                continue
        return out
```

```python
#!/usr/bin/env python3
"""refresh_all — 本地一把刷三源(竞彩 sporttery + 欧盘 500翻页 + Poly 7897代理)
→ 写本地 odds_watch 缓存 → 按 label 对齐到 zucai_num、算变化+分歧 → POST HK 看板
两端点(/api/ingest/zucai 竞彩raw, /api/ingest/odds 赔率面板)。取代 collect_zucai。

用法:
  python3 tools/refresh_all.py --once          # 跑一次
  python3 tools/refresh_all.py --loop 300        # 每300s
  python3 tools/refresh_all.py --once --dry-run  # 抓+对齐但不推 HK(冒烟)
环境: WC_INGEST_URL / WC_INGEST_PW / WC_INGEST_USER(admin) / WC_HTTPS_PROXY(7897)
"""
from __future__ import annotations
import argparse, base64, json, os, sys, time, urllib.request

REPO = os.path.dirname(os.path.dirname(os.path.abspath(__file__)))
sys.path.insert(0, os.path.join(REPO, "tools"))
sys.path.insert(0, REPO)
import odds_watch as ow
import poly_fetch_hk as pf
from backend import sporttery

PROXY = os.environ.get("WC_HTTPS_PROXY", "http://127.0.0.1:7897")
INGEST = os.environ.get("WC_INGEST_URL", "http://18.166.71.60:8000").rstrip("/")
PW = os.environ.get("WC_INGEST_PW", "")
USER = os.environ.get("WC_INGEST_USER", "admin")

def arrow(old, new):
    if old is None or new is None:
        return "-"
    return "▲" if new > old else ("▼" if new < old else "-")

def _devig_from_had(had):
    """1X2 欧赔 → 去水% {h,d,a}; 缺则 None。"""
    if not had or any(had.get(k) in (None, 0) for k in ("h", "d", "a")):
        return None
    imp = {k: 1.0 / had[k] for k in ("h", "d", "a")}
    s = sum(imp.values())
    return {k: round(imp[k] / s * 100, 1) for k in ("h", "d", "a")}

def _delta_had(prev, new_had):
    old = (prev or {}).get("had") or {}
    return {k: arrow(old.get(k), (new_had or {}).get(k)) for k in ("h", "d", "a")}

def _delta_devig(prev_key, prev, new_devig):
    old = (prev or {}).get(prev_key) or {}
    return {k: arrow(old.get(k), (new_devig or {}).get(k)) for k in ("h", "d", "a")}

def build_panel(zucai_items, consensus_items, poly_items, prev_lookup):
    by_label_cons = {it["label"]: it for it in consensus_items}
    by_label_poly = {it["label"]: it for it in poly_items}
    panel = []
    for z in zucai_items:
        mk, label = z["match_key"], z["label"]
        zp = z.get("payload") or {}
        had = zp.get("had")
        z_devig = _devig_from_had(had)
        src = {"zucai": {"had": had, "hhad": zp.get("hhad"), "ttg": zp.get("ttg"),
                          "devig": z_devig, "delta": _delta_had(prev_lookup("zucai", mk), had),
                          "stale": had is None}}
        # consensus(按 label 对齐)
        c = by_label_cons.get(label)
        if c:
            cp = c.get("payload") or {}
            c_devig = cp.get("devig_pct")
            src["consensus"] = {"euro": cp.get("had"), "devig": c_devig, "n_books": cp.get("n_books"),
                                 "delta": _delta_devig("devig_pct", prev_lookup("consensus", c["match_key"]), c_devig),
                                 "stale": c_devig is None}
        else:
            src["consensus"] = {"stale": True}
        # poly(按 label 对齐; 缓存里其实同 zucai_num, label 兜底)
        p = by_label_poly.get(label)
        if p:
            pp = p.get("payload") or {}
            p_devig = pp.get("poly_devig")
            src["poly"] = {"devig": p_devig,
                            "delta": _delta_devig("poly_devig", prev_lookup("poly", mk), p_devig),
                            "stale": p_devig is None}
        else:
            src["poly"] = {"stale": True}
        # divergence: 竞彩去水 − 欧盘去水 (pp)
        cdv = src["consensus"].get("devig") if not src["consensus"].get("stale") else None
        div = ({k: round((z_devig[k] - cdv[k]), 1) for k in ("h", "d", "a")}
               if (z_devig and cdv) else {"h": None, "d": None, "a": None})
        panel.append({"match_key": mk, "label": label, "ko": z.get("ko"),
                      "fetched_at": ow.now_bj(), "sources": src, "divergence": div})
    return panel

def _post(path, body):
    data = json.dumps(body).encode()
    hdr = {"Content-Type": "application/json"}
    if PW:
        hdr["Authorization"] = "Basic " + base64.b64encode(f"{USER}:{PW}".encode()).decode()
    req = urllib.request.Request(INGEST + path, data=data, headers=hdr, method="POST")
    try:
        with urllib.request.urlopen(req, timeout=30) as r:
            return json.load(r)
    except Exception as e:  # noqa: BLE001
        sys.stderr.write(f"  [warn] POST {path} -> {e!r}\n")
        return None

def run_once(dry_run=False):
    conn = ow.connect(ow.DEFAULT_CACHE)
    # 上一快照(算 delta), 在 ingest 前取
    def prev_lookup(source, match_key):
        return ow.latest(conn, source, match_key)
    # 1) 抓三源
    raw_env = sporttery.fetch({})                      # 竞彩 raw 信封(推 /api/ingest/zucai)
    zucai_items = ow.fetch_zucai()
    try:
        consensus_items = ow.fetch_consensus()
    except Exception as e:  # noqa: BLE001
        sys.stderr.write(f"  [warn] consensus 失败: {e!r}\n"); consensus_items = []
    poly_items = _fetch_poly_local()
    # 2) 组面板(用 ingest 前的 prev 算 delta)
    panel = build_panel(zucai_items, consensus_items, poly_items, prev_lookup)
    # 3) 写本地缓存(供本地分析 + 下轮 delta)
    for src, items in (("zucai", zucai_items), ("consensus", consensus_items), ("poly", poly_items)):
        for it in items:
            ow.save(conn, src, it["match_key"], it["label"], it.get("ko", ""), it["payload"])
    conn.commit()
    print(f"[{ow.now_bj()}] 三源: 竞彩{len(zucai_items)} 欧盘{len(consensus_items)} Poly{len(poly_items)}; 面板{len(panel)}场")
    # 4) 推 HK
    if dry_run:
        print("  --dry-run: 不推 HK"); return
    r1 = _post("/api/ingest/zucai", raw_env)
    r2 = _post("/api/ingest/odds", {"items": panel})
    print(f"  推送: zucai={r1} odds={r2}")

def _fetch_poly_local():
    """本地代理跑 poly build-list→fetch→读 ingest items。失败返 []。"""
    import tempfile
    pf.set_proxy(PROXY)
    try:
        ml = os.path.join(tempfile.gettempdir(), "wc_poly_matches.json")
        pi = os.path.join(tempfile.gettempdir(), "wc_poly_ingest.json")
        pf.build_list(out=ml)            # 见下: build-list 入口函数名以实际为准
        pf.fetch(ml, pi, proxy=PROXY)
        return json.load(open(pi, encoding="utf-8"))
    except Exception as e:  # noqa: BLE001
        sys.stderr.write(f"  [warn] poly 本地抓失败: {e!r}\n"); return []

if __name__ == "__main__":
    ap = argparse.ArgumentParser()
    ap.add_argument("--once", action="store_true")
    ap.add_argument("--loop", type=int, metavar="SEC")
    ap.add_argument("--dry-run", action="store_true")
    a = ap.parse_args()
    if a.loop:
        while True:
            run_once(a.dry_run); time.sleep(a.loop)
    else:
        run_once(a.dry_run)
```

```javascript
function oddsArrow(sym) {
  if (sym === "▲") return el("span", "od-up", "▲");      // 升水(红)
  if (sym === "▼") return el("span", "od-down", "▼");    // 降水(绿)
  return el("span", "od-flat", "");
}

function renderOddsPanel(odds) {
  const box = el("div", "odds-panel");
  const head = el("div", "odds-head", "赔率(三源)");
  box.appendChild(head);
  if (!odds || !odds.sources) { box.appendChild(el("div", "odds-empty", "赔率待刷新")); return box; }
  const body = el("div", "odds-body");
  const s = odds.sources;
  // 三源 × 主/平/负 表
  const rows = [
    ["竞彩", s.zucai, "had", false],
    ["欧盘", s.consensus, "devig", true],
    ["Poly", s.poly, "devig", true],
  ];
  const tbl = el("div", "odds-tbl");
  tbl.appendChild(el("div", "ot-h", ""));
  for (const c of ["主", "平", "负"]) tbl.appendChild(el("div", "ot-h", c));
  for (const [name, src, key, isPct] of rows) {
    tbl.appendChild(el("div", "ot-name", name));
    if (!src || src.stale) { for (const k of ["h","d","a"]) tbl.appendChild(el("div", "ot-cell muted", "—")); continue; }
    const vals = src[key] || {};
    const dl = src.delta || {};
    for (const k of ["h", "d", "a"]) {
      const cell = el("div", "ot-cell");
      const v = vals[k];
      cell.appendChild(el("span", null, v == null ? "—" : (isPct ? fmtNum(v, 1) + "%" : fmtNum(v, 2))));
      cell.appendChild(oddsArrow(dl[k]));
      tbl.appendChild(cell);
    }
  }
  body.appendChild(tbl);
  // 让球 / 总进球(竞彩)
  if (s.zucai && s.zucai.hhad) {
    const h = s.zucai.hhad;
    body.appendChild(el("div", "odds-extra", `让球(${h.line > 0 ? "+" : ""}${h.line}): 主 ${fmtNum(h.h)} / 客 ${fmtNum(h.a)}`));
  }
  if (s.zucai && s.zucai.ttg && Object.keys(s.zucai.ttg).length) {
    const ttg = s.zucai.ttg;
    const parts = Object.keys(ttg).slice(0, 6).map((g) => `${g}球 ${fmtNum(ttg[g])}`);
    body.appendChild(el("div", "odds-extra", "总进球: " + parts.join(" / ")));
  }
  // 分歧徽标
  const dv = odds.divergence || {};
  const big = ["h", "d", "a"].some((k) => dv[k] != null && Math.abs(dv[k]) >= 3);
  if (big) body.appendChild(el("div", "odds-diverge", "⚠ 竞彩偏离欧盘 ≥3pp"));
  if (odds.fetched_at) body.appendChild(el("div", "odds-ts", "抓取 " + esc(String(odds.fetched_at)).slice(11, 16)));
  // 折叠
  head.addEventListener("click", () => box.classList.toggle("open"));
  box.appendChild(body);
  return box;
}
```
